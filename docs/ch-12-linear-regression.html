coefficient of determination \(r^2\)

# Example 12.4, 12.9
x <- c(132, 129, 120, 113.2, 105, 92, 84, 83.2, 88.4, 59, 80, 81.5, 71, 69.2)
y <- c(46, 48, 51, 52.1, 54, 52, 59, 58.7, 61.6, 64, 61.4, 54.6, 58.8, 58)

mod <- lm(y ~ x)

SSE <- anova(mod)[2,2]
SST <- anova(mod)[1,2] + anova(mod)[2,2]
1 - (SSE/SST)
## [1] 0.7907602

Or (simply):

cor(x,y)^2
## [1] 0.7907602
