0.5 1.3 Measures of location

prices <- c(379, 425, 450, 450, 499, 529, 535, 535, 545, 599, 665,
            675, 699, 699, 725, 725, 745, 799)

mean(prices)
## [1] 593.2222
median(prices)
## [1] 572
## quartiles
quantile(prices)
##    0%   25%   50%   75%  100% 
## 379.0 506.5 572.0 699.0 799.0
## trimmed mean
mean(prices, trim = .1) ## 10% trimmed mean
## [1] 593.75

0.6 1.4 Measures of variability

Sample variance

prices <- c(379, 425, 450, 450, 499, 529, 535, 535, 545, 599, 665,
            675, 699, 699, 725, 725, 745, 799)

var(prices)
## [1] 15981.48

Sample standard deviation

sqrt(var(prices))
## [1] 126.4179
sd(prices)
## [1] 126.4179

Five number summary

(min, lower-hinge, median, upper-hinge, max)

fivenum(prices)
## [1] 379 499 572 699 799
